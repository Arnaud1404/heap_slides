\section{Le Cœur : 1 Type d'Exploit = 1 Primitive (12 min)}

\begin{frame}[fragile]{Use-After-Free (UAF) \& Safe Linking (1/3)}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Vulnérabilité} : \textit{Use-After-Free}
            \begin{itemize}
                \item \textbf{Objectif} : Détourner la liste (Tcache).
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Conséquence globale}
            \begin{itemize}
                \item \textit{Arbitrary Write} via \texttt{malloc}.
            \end{itemize}
        \end{column}
    \end{columns}
    \vspace{-1.5em}

    \begin{ccode}
        long *a = malloc(0x20); free(a); // 'a' dans Tcache
        *a = 0xdeadbeef;                 // UAF : fd corrompu !
        malloc(0x20);                    // Dépile 'a'
        long *target = malloc(0x20);     // Ciblera 0xdeadbeef !
    \end{ccode}
    \vspace{-1.5em}

    \begin{pwndbg}[tcachebins]{text}
        tcachebins
        0x20 [  2]: 0x5...010 @\textcolor{pwndbgblue}{->}@ @\textcolor{pwndbgred}{0xdeadbeef}@
    \end{pwndbg}
\end{frame}

\begin{frame}[fragile]{Use-After-Free (UAF) \& Safe Linking (2/3)}
    \textbf{Mitigation : Safe Linking (GLIBC 2.32+)}
    \begin{itemize}
        \item Masquage cryptographique du Tcache Poisoning aveugle.
    \end{itemize}
    \vspace{0.1em}

    \begin{columns}[T]
        \begin{column}{0.50\textwidth}
            \textbf{Formule algorithmique}
            \begin{itemize}
                \item $C = (L \gg 12) \oplus P$
                \item $L$ : L'adresse stockant le pointeur.
                \item $P$ : Le vrai pointeur ciblé.
                \item $C$ : Valeur stockée en RAM.
            \end{itemize}
        \end{column}

        \begin{column}{0.46\textwidth}
            \textbf{L'impact sur l'attaquant}
            \begin{itemize}
                \item Forcer un pointeur provoque un crash (\texttt{SIGSEGV}).
                \item Protège les allocations.
                \item \textbf{Nécessite} : Un \textit{Heap Leak}.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}[fragile]{Heap Overflow \& Overlapping}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Mécanisme} (Cible : \texttt{size})
            \begin{itemize}
                \item Extension d'un chunk vers un autre.
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Conséquence} (\textit{Aliasing})
            \begin{itemize}
                \item Effacement de données adjacentes.
            \end{itemize}
        \end{column}
    \end{columns}
    \vspace{-1.5em}

    \begin{ccode}
        long *p1 = malloc(0x80-8); long *p2 = malloc(0x500-8);
        long *p3 = malloc(0x80-8); malloc(0x10); // guard

        *(p2 - 1) = 0x581; // OVERFLOW (size altée)
        free(p2); // p2 dans l'Unsorted Bin avec fausse taille

        long *p4 = malloc(0x580 - 8); // Chevauche p3 !
        memset(p4, '4', 0x580 - 8);   // Écrase p3 en silence
    \end{ccode}
    \vspace{-1.5em}

    \begin{pwndbg}[heap]{text}
        0x5...000 0x...081 (p1)
        0x5...080 @\textcolor{pwndbgred}{0x0...581}@ (p2 corrompu, taille \textit{fake} 0x580)
        0x5...580 0x0...081 (p3 originel non modifié)
        ...
        0x5...600 0x3434343434343434 (p4 et p3 tapent ici !)
    \end{pwndbg}
\end{frame}

        \end{column}

    \begin{ccode}
        typedef struct { void (*execute)(); } task_t; // 8 octets
        typedef struct { char content[8]; } msg_t;    // 8 octets

        task_t *task = malloc(sizeof(task_t)); task->execute = lose;
        free(task); // UAF: dangling pointer !

        msg_t *msg = malloc(sizeof(msg_t)); // Allocation même taille
        *(long *)msg->content = (long)win;  // Type Confusion

        task->execute(); // Exécute win() !
    \end{ccode}

\begin{frame}{Évolutions : GLIBC 2.42 \& 2.43}
    \textbf{GLIBC 2.42 (Août 2025)}
    \vspace{-0.2cm}
    \begin{itemize}
        \item Cohérence exigée à l'insertion en \textit{Unsorted} / \textit{Largebins}.
        \item \textbf{Cible} : Neutralisation des \textit{Largebin Attacks}.
        \item Règle stricte bilatérale : \texttt{P->fd\_nextsize->bk\_nextsize == P}
    \end{itemize}
    \vspace{0.1cm}

    \textbf{GLIBC 2.43 (Janvier 2026)}
    \vspace{-0.2cm}
    \begin{itemize}
        \item \textbf{Refonte} : Suppression des \textit{Fastbins} (trop de failles historiques).
        \item Petites allocations gérées par \textit{Tcache} et \textit{Smallbins}.
        \item \textit{Safe Linking} étendu à tout l'allocateur.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Conclusion}
    \textbf{Le crépuscule des métadonnées ?}
    \vspace{-0.2cm}
    \begin{itemize}
        \item Durcissements successifs : \textit{Safe Linking}, vérifications croisées (\texttt{nextsize}).
        \item Disparition des listes peu protégées (\textit{Fastbins}).
        \item Arbitrary Write devient coûteux (multiples fuites nécessaires : `heap`, `libc`).
    \end{itemize}
    \vspace{0.4cm}

    \textbf{Pivot vers la corruption applicative}
    \vspace{-0.2cm}
    \begin{itemize}
        \item Les défenses modernes protègent la machinerie GLIBC, pas la logique métier.
        \item \textbf{Data-Only Attacks} : Attaquer directement les structures adjacentes (privilèges, états).
        \item \texttt{malloc} n'est plus la cible, mais l'outil.
    \end{itemize}
\end{frame}
