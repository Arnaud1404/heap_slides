\begin{frame}[fragile]{Use-After-Free (UAF) \& Safe Linking}
    \textbf{Vulnérabilité \& Primitive}
    \begin{itemize}
        \item \textbf{UAF} : accès pointeur post-libération.
        \item \textbf{Tcache Poisoning} : corruption pointeur de chaînage \texttt{fd}.
        \item Objectif final : point d'écriture arbitraire via \texttt{malloc()}.
    \end{itemize}
    
    \textbf{Mitigation : Safe Linking (GLIBC 2.32+)}
    \begin{itemize}
        \item Masquage cryptographique \textit{freelist} : $C = (L \gg 12) \oplus P$.
