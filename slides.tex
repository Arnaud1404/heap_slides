\section{Les techniques d'exploitation}

\begin{frame}{Les bases d'un exploit}

    \begin{itemize}
        \item Enchaînement de techniques
        \item \textbf{3 bugs} + \textbf{feng shui}
        \item \textit{Feng shui} : placement des chunks, ordre des allocations, ordre des frees
        \item UAF : pointeur gardé après \texttt{free}
        \item Double free : même chunk libéré deux fois
        \item Heap overflow : débordement sur le chunk voisin
    \end{itemize}

\end{frame}

\begin{frame}[fragile]{Défense : Safe Linking (GLIBC 2.32+)}
    \textbf{Mitigation : Chiffrement des pointeurs}
    \begin{itemize}
        \item Masquage du Tcache Poisoning "aveugle".
    \end{itemize}
    \vspace{0.1em}

    \begin{columns}[T]
        \begin{column}{0.50\textwidth}
            \textbf{Formule de protection}
            \begin{itemize}
                \item $C = (L \gg 12) \oplus P$
                \item $L$ : Adresse de stockage (Leak requis).
                \item $P$ : Le vrai pointeur (Destination).
                \item $C$ : Valeur stockée (Mélangée).
            \end{itemize}
        \end{column}

        \begin{column}{0.46\textwidth}
            \textbf{Impact Offensif}
            \begin{itemize}
                \item Écrire \textit{brut} une adresse provoque un crash (\texttt{SIGSEGV}).
                \item Rend les exploits non-déterministes.
                \item \textbf{Solution} : Fuite (Heap Leak).
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}[fragile]{Exploitation Applicative : Type Confusion}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Le Pivot Applicatif}
            \begin{itemize}
                \item GLIBC ignorée.
                \item Cible : Logique interne du programme.
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Avantage Majeur}
            \begin{itemize}
                \item Aucune sécurité type \textit{Safe Linking}.
            \end{itemize}
        \end{column}
    \end{columns}
    \vspace{0.2em}
    \begin{ccode}
        typedef struct { void (*execute)(); } task_t; // 8 octets
        typedef struct { char content[8]; } msg_t;    // 8 octets

        task_t *task = malloc(sizeof(task_t)); task->execute = lose;
        free(task); // UAF: dangling pointer !

        msg_t *msg = malloc(sizeof(msg_t)); // Allocation même taille
        *(long *)msg->content = (long)win;  // Confusion de type

        task->execute(); // Exécute win() !
    \end{ccode}
\end{frame}



\begin{frame}{Évolutions : GLIBC 2.42 \& 2.43}
    \textbf{GLIBC 2.42 (Août 2025)}
    \vspace{-0.2cm}
    \begin{itemize}
        \item Cohérence exigée à l'insertion en \textit{Unsorted} / \textit{Largebins}.
        \item \textbf{Cible} : Neutralisation des \textit{Largebin Attacks}.
        \item Vérification d'intégrité : \texttt{P->bk\_nextsize->fd\_nextsize == P}
    \end{itemize}
    \vspace{0.1cm}

    \textbf{GLIBC 2.43 (Janvier 2026)}
    \vspace{-0.2cm}
    \begin{itemize}
        \item \textbf{Refonte} : Suppression des \textit{Fastbins} (trop de failles historiques).
        \item Petites allocations gérées par \textit{Tcache} et \textit{Smallbins}.
        \item \textit{Safe Linking} étendu à tout l'allocateur.
    \end{itemize}
\end{frame}
