\newtcblisting{pwndbg}[2][]{%
    listing engine=minted,
    minted style=xcode,
    minted language=#2,
    minted options={fontsize=\scriptsize, breaklines, escapeinside=@@},
    colback=white,
    colframe=black, % Contraste max
    colupper=black,
    boxrule=1pt,
    arc=0pt, % Coins carrés plus "terminal"

    drop shadow=black!30!white,

    fonttitle=\bfseries\ttfamily\scriptsize,
    coltitle=black,
    colbacktitle=gray!20,
    listing only,
    enhanced,
    breakable=true
}
